- Write a program to<span style="color:rgb(255, 31, 31)"> Calculate Number Of Days, Hours, Minutes and Seconds In a Month</span> : 

```cpp
#include <iostream>

using namespace std;

short ReadYear()
{
	short Year;

	cout << "Please Enter a Year to check ?\n";
	cin >> Year;

	return Year;
}

short ReadMonth()
{
	short Month;

	cout << "Please Enter a Month to check ?\n";
	cin >> Month;

	return Month;
}

bool IsLeapYear(short Year)
{
	// if year is divisible by 4 AND not divisible by 100
	// OR if year is divisible by 400
	// then it is a leap year
	return (Year % 4 == 0 && Year % 100 == 0 || Year % 400 == 0);
}

short NumberOfDaysInAMonth(short Month, short Year)
{
	if (Month < 1 || Month > 12)
	{
		return 0;
	}
	
	if (Month == 2)
	{
		return IsLeapYear(Year) ? 29 : 28;
	}

	short arr31Days[7] = { 1,3,5,7,8,10,12 };

	for (short i = 1; i <= 7; i++)
	{
		if (arr31Days[i - 1] == Month)
		{
			return 31;
		}

	}
	//if you reach here then its 30 days.
	return 30;
}

short NumberOfHoursInAMonth(short Month, short Year)
{
	return NumberOfDaysInAMonth(Month, Year) * 24;
}

int NumberOfMinutesInAYear(short Month, short Year)
{
	return NumberOfHoursInAMonth(Month, Year) * 60;
}

int NumberOfSecondsInAYear(short Month, short Year)
{
	return NumberOfMinutesInAYear(Month, Year) * 60;
}

int	main()
{

	short Year = ReadYear();
	short Month = ReadMonth();

	cout << "\nNumber of Days in Month [" << Month << "] : "
		<< NumberOfDaysInAMonth(Month, Year) << endl;
	
	cout << "\nNumber of Hours in Month [" << Month << "] : "
		<< NumberOfHoursInAMonth(Month, Year) << endl;

	cout << "\nNumber of Minutes in Month [" << Month << "] : "
		<< NumberOfMinutesInAYear(Month, Year) << endl;

	cout << "\nNumber of Seconds in Month [" << Month << "] : "
		<< NumberOfSecondsInAYear(Month, Year) << endl;


	system("pause>0");

	return 0;
}
```
