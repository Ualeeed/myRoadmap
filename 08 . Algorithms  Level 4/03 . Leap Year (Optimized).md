- Re-write the Leap Year program as short as you can ( <span style="color:rgb(255, 31, 31)">optimize the code</span>) :


```cpp
#include <iostream>

using namespace std;

short ReadYear()
{
	short Year;

	cout << "Please Enter a Year to check ?\n";
	cin >> Year;

	return Year;
}

bool IsLeapYear(short Year)
{
	// if year is divisible by 4 AND not divisible by 100
	// OR if year is divisible by 400
	// then it is a leap year
	return (Year % 4 == 0 && Year % 100 == 0 || Year % 400 == 0);
}

int	main()
{

	short Year = ReadYear();

	if (IsLeapYear(Year))
	{
		cout << "Yes, Year [" << Year << "] is a Leap Year\n";
	}
	else
	{
		cout << "No, Year [" << Year << "] is not a Leap Year\n";
	}

	system("pause>0");

	return 0;
}
```
