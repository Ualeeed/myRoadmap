- Write a program to <span style="color:rgb(255, 31, 31)">check if the Year is a Leap year or not</span> : 
- <span style="color:rgb(255, 31, 31)">Leap Year</span> : <span style="color:rgb(255, 31, 31)">سنة كبيسة</span>
- بمعنى ان ديك السنة غيكون فيها 366 يوم عوض 365.

```cpp
#include <iostream>

using namespace std;

short ReadYear()
{
	short Year;

	cout << "Please Enter a Year to check ?\n";
	cin >> Year;

	return Year;
}

bool IsLeapYear(short Year)
{
	// leap year if perfectly divisible by 400
	if (Year % 400 == 0)
	{
		return true;
	}
	// not a leap year if divisible by 100
	// but not divisible by 400
	else if (Year % 100 == 0)
	{
		return false;
	}
	// leap year if not divisible by 100
	// but divisible by 4
	else if (Year % 4 == 0)
	{
		return true;
	}
	// all other years are not leap years
	else
	{
		return false;
	}

}

int	main()
{

	short Year = ReadYear();

	if (IsLeapYear(Year))
	{
		cout << "Yes, Year [" << Year << "] is a Leap Year\n";
	}
	else
	{
		cout << "No, Year [" << Year << "] is not a Leap Year\n";
	}

	system("pause>0");

	return 0;
}
```
