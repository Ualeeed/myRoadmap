```
The solution we add [cin.ignore(1, '\n');] before getline(cin, Name); 
```

```cpp
#include <iostream>
#include <string>
using namespace std;


int main()
{
	int Number;
	string Name;
	string Country;

	cout << "Please enter your employer Number : " << endl;
	cin >> Number;

	cout << "Please enter your Name : " << endl;
	cin.ignore(1, '\n');
	getline(cin, Name);

	cout << "Please enter your Country : " << endl;
	cin >> Country;


	cout << "employer Number : " << Number << endl;
	cout << "Your Name is : " << Name << endl;
	cout << "Your Country is : " << Country << endl;


	return 0;
}
```
