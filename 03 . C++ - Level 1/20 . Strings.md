## What is a String?

A string variable contains a <mark style="background: #BBFABBA6;">collection of characters</mark> surrounded by double quotes. **String** is an <mark style="background: #FFB86CA6;">Array of Characters</mark>, but it's much more powerful than a simple character array.

## Basic String Operations

### Getting String Length
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string myString = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    
    cout << "Length using .length(): " << myString.length() << endl;
    cout << "Length using .size(): " << myString.size() << endl;  // Same as length()
    cout << "Is empty?: " << myString.empty() << endl;
    
    return 0;
}
```

### Accessing Characters
> **Important**: In <mark style="background: #FFB8EBA6;">C++</mark> counting in **String**, **arrays**, and **Enums** always starts from 0, 1, 2...

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string myString = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    
    // Method 1: Using [] operator (no bounds checking)
    cout << "Character at index 2: " << myString[2] << endl;
    
    // Method 2: Using at() method (with bounds checking)
    cout << "Character at index 2 (safe): " << myString.at(2) << endl;
    
    // Getting first and last characters
    cout << "First character: " << myString.front() << endl;
    cout << "Last character: " << myString.back() << endl;
    
    return 0;
}
```

## String Concatenation

### Basic Concatenation
**This method is called "concatenation":**
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string s1 = "10", s2 = "20";
    string s3 = s1 + s2;  // Result: "1020"
    
    cout << "Concatenation: " << s3 << endl;
    
    // Multiple ways to concatenate
    string greeting = "Hello, ";
    string name = "World!";
    
    // Method 1: Using + operator
    string message1 = greeting + name;
    
    // Method 2: Using += operator
    string message2 = greeting;
    message2 += name;
    
    // Method 3: Using append() method
    string message3 = greeting;
    message3.append(name);
    
    cout << "Message 1: " << message1 << endl;
    cout << "Message 2: " << message2 << endl;
    cout << "Message 3: " << message3 << endl;
    
    return 0;
}
```

## String to Number Conversion

If you want to calculate numbers inside strings, you need to convert them:

```cpp
#include <iostream>
#include <string>
#include <stdexcept>
using namespace std;

int main() {
    string s1 = "10", s2 = "20";
    
    try {
        // Convert strings to numbers and calculate
        int sum = stoi(s1) + stoi(s2);  // Result: 30
        cout << "Sum: " << sum << endl;
        
        // Other conversion functions
        string floatStr = "3.14";
        string doubleStr = "2.718281828";
        
        float f = stof(floatStr);
        double d = stod(doubleStr);
        
        cout << "Float: " << f << endl;
        cout << "Double: " << d << endl;
        
 
    return 0;
}
```

## Reading Strings with Spaces

### Using getline() for Complete Lines
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string name, city, country;
    int age;
    float monthlySalary;
    char gender;
    bool isMarried;

    cout << "Please enter your name: ";
    getline(cin, name);  // Reads entire line including spaces

    cout << "Please enter your age: ";
    cin >> age;
    
    // Clear the input buffer after reading numbers
    cin.ignore();  // Important for mixed input!
    
    cout << "Please enter your city: ";
    getline(cin, city);
    
    cout << "Please enter your country: ";
    getline(cin, country);
    
    cout << "Please enter your monthly salary: ";
    cin >> monthlySalary;
    
    cout << "Please enter your gender (M/F): ";
    cin >> gender;
    
    cout << "Are you married? (1/0): ";
    cin >> isMarried;

    cout << "\n=== Personal Information ===" << endl;
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;
    cout << "City: " << city << endl;
    cout << "Country: " << country << endl;
    cout << "Monthly Salary: $" << monthlySalary << endl;
    cout << "Yearly Salary: $" << monthlySalary * 12 << endl;
    cout << "Gender: " << gender << endl;
    cout << "Married: " << (isMarried ? "Yes" : "No") << endl;

    return 0;
}
```

## Advanced String Operations

### Substring Operations
The substr() function returns a newly constructed string object with its value initialized to a copy of a substring, starting at a character position and spanning a specified length.

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string text = "Hello, World! Welcome to C++";
    
    // Extract substrings
    string hello = text.substr(0, 5);        // "Hello"
    string world = text.substr(7, 5);        // "World"
    string welcome = text.substr(14, 7);     // "Welcome"
    string fromPos = text.substr(14);        // "Welcome to C++" (to end)
    
    cout << "Original: " << text << endl;
    cout << "Hello: " << hello << endl;
    cout << "World: " << world << endl;
    cout << "Welcome: " << welcome << endl;
    cout << "From position 14: " << fromPos << endl;
    
    return 0;
}
```

### Finding and Searching
```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string text = "The quick brown fox jumps over the lazy dog";
    
    // Find substring
    size_t pos = text.find("fox");
    if (pos != string::npos) {
        cout << "Found 'fox' at position: " << pos << endl;
    }
    
    // Find last occurrence
    size_t lastPos = text.rfind("the");
    cout << "Last 'the' at position: " << lastPos << endl;
    
    // Find any character from a set
    size_t vowelPos = text.find_first_of("aeiou");
    cout << "First vowel at position: " << vowelPos << endl;
    
    // Check if string contains substring (C++23)
    // bool contains = text.contains("fox");  // C++23 feature
    
    return 0;
}
```

### String Replacement
We can use std::string::find() to find the position of occurrence of the substring and std::string::replace() to replace that part of the string.

```cpp
#include <iostream>
#include <string>
using namespace std;

// Function to replace all occurrences
void replaceAll(string& str, const string& from, const string& to) {
    size_t start_pos = 0;
    while ((start_pos = str.find(from, start_pos)) != string::npos) {
        str.replace(start_pos, from.length(), to);
        start_pos += to.length();
    }
}

int main() {
    string text = "Hello World! Hello Universe!";
    
    // Replace single occurrence
    size_t pos = text.find("Hello");
    if (pos != string::npos) {
        text.replace(pos, 5, "Hi");  // Replace first "Hello" with "Hi"
    }
    cout << "After single replace: " << text << endl;
    
    // Replace all occurrences using our function
    string text2 = "Hello World! Hello Universe!";
    replaceAll(text2, "Hello", "Hi");
    cout << "After replace all: " << text2 << endl;
    
    return 0;
}
```

### Comparison Operations
```cpp
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string str1 = "Apple";
    string str2 = "apple";
    string str3 = "Apple";
    
    // Case-sensitive comparison
    cout << "str1 == str3: " << (str1 == str3) << endl;  // true
    cout << "str1 == str2: " << (str1 == str2) << endl;  // false
    
    // Lexicographical comparison
    cout << "str1 < str2: " << (str1 < str2) << endl;   // true (A < a in ASCII)
    
    // Case-insensitive comparison (custom function)
    auto caseInsensitiveCompare = [](const string& a, const string& b) {
        string lowerA = a, lowerB = b;
        transform(lowerA.begin(), lowerA.end(), lowerA.begin(), ::tolower);
        transform(lowerB.begin(), lowerB.end(), lowerB.begin(), ::tolower);
        return lowerA == lowerB;
    };
    
    cout << "Case-insensitive comparison: " 
         << caseInsensitiveCompare(str1, str2) << endl;  // true
    
    return 0;
}
```
