### Function Overloading

Function overloading in C++ allows us to define multiple functions with the same name but with different parameters. The compiler determines which function to call based on the arguments passed to it.

```cpp
#include <iostream>
using namespace std;

// Function overloading examples
int add(int a, int b) {
    return a + b;
}

double add(double a, double b) {
    return a + b;
}

int add(int a, int b, int c) {
    return a + b + c;
}

int main() {
    cout << add(5, 3) << endl;        // Calls int version
    cout << add(2.5, 3.7) << endl;   // Calls double version
    cout << add(1, 2, 3) << endl;    // Calls three-parameter version
    
    return 0;
}
```

### Function Parameters - Pass by Value vs Pass by Reference

```cpp
#include <iostream>
using namespace std;

// Pass by value (creates a copy)
void changeValue(int x) {
    x = 100;
    cout << "Inside function: " << x << endl;
}

// Pass by reference (works with original variable)
void changeReference(int& x) {
    x = 100;
    cout << "Inside function: " << x << endl;
}

// Pass by pointer
void changePointer(int* x) {
    *x = 100;
    cout << "Inside function: " << *x << endl;
}

int main() {

    int num = 50;
    
    cout << "Original value: " << num << endl;
    
    // Pass by value - original doesn't change
    changeValue(num);
    cout << "After pass by value: " << num << endl;
    
    // Pass by reference - original changes
    changeReference(num);
    cout << "After pass by reference: " << num << endl;
    
    // Reset for pointer example
    num = 50;
    
    // Pass by pointer - original changes
    changePointer(&num);
    cout << "After pass by pointer: " << num << endl;
    
    return 0;
}
```

### Default Parameters

```cpp
#include <iostream>
using namespace std;

// Function with default parameters
void greetUser(string name, string greeting = "Hello", int times = 1) {
    for(int i = 0; i < times; i++) {
        cout << greeting << ", " << name << "!" << endl;
    }
}

int main() {
    greetUser("Alice");                    // Uses defaults
    greetUser("Bob", "Hi");               // Custom greeting, default times
    greetUser("Charlie", "Welcome", 3);    // All custom parameters
    
    return 0;
}
```

### Inline Functions

```cpp
#include <iostream>
using namespace std;

// Inline function - suggests compiler to replace function call with function body
inline int square(int x) {
    return x * x;
}

// Better for small, frequently called functions
inline int max(int a, int b) {
    return (a > b) ? a : b;
}

int main() {
    int num = 5;
    cout << "Square of " << num << " is " << square(num) << endl;
    cout << "Max of 10 and 20 is " << max(10, 20) << endl;
    
    return 0;
}
```

### Recursive Functions

```cpp
#include <iostream>
using namespace std;

// Recursive function example - Factorial
int factorial(int n) {
    // Base case
    if (n <= 1) {
        return 1;
    }
    // Recursive case
    return n * factorial(n - 1);
}

// Recursive function example - Fibonacci
int fibonacci(int n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() {
    cout << "Factorial of 5: " << factorial(5) << endl;
    cout << "10th Fibonacci number: " << fibonacci(10) << endl;
    
    return 0;
}
```





1. **Keep functions small and focused** - Each function should do one thing well
2. **Use meaningful names** - Function names should clearly describe what they do
3. **Minimize parameters** - Too many parameters make functions hard to use
4. **Use const when appropriate** - Especially for reference parameters that shouldn't be modified
5. **Document complex functions** - Add comments explaining the purpose and parameters
6. **Prefer pass by reference for large objects** - Avoid unnecessary copying
7. **Use default parameters wisely** - They can make functions more flexible but also more complex
