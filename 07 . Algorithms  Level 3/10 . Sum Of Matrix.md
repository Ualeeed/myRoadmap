#include <iostream>
#include <string>
#include <iomanip>

using namespace std;


int RandomNumber(int From, int To)
{
	//Function to generate a random number
	int randNum = rand() % (To - From + 1) + From;
	return randNum;
}


void FillMatrixWithRandomNumbers(int arr[3][3], short Rows, short
	Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			arr[i][j] = RandomNumber(1, 10);
		}
	}
}

void PrintMatrix(int arr[3][3], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			printf(" %0*d ", 2, arr[i][j]);
			//cout << setw(3) << arr[i][j] << " ";
		}
		cout << "\n";
	}
}

int SumOfMatrix(int Matrix1[3][3], short Rows, short Cols)
{
	int Sum = 0;

	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			Sum += Matrix1[i][j];
		}
	}
	return Sum;
}


int main()
{
	//Seeds the random number generator in C++, called only once
	srand((unsigned)time(NULL));


	int Matrix1[3][3];
	FillMatrixWithRandomNumbers(Matrix1, 3, 3);
	cout << "\nMatrix1:\n";


	PrintMatrix(Matrix1, 3, 3);


	cout << "\nSum of Matrix1 is: " << SumOfMatrix(Matrix1, 3, 3);



	system("pause>0");
}

## Explanation

This program calculates and displays the sum of all elements in a 3x3 matrix.

### Key Concepts:

**Matrix Sum:**
- Sum of matrix = sum of all individual elements
- Requires traversing every element once using nested loops

### Algorithm Breakdown:

1. **Matrix Generation:** Creates a 3x3 matrix with random numbers (1-10)

2. **Sum Calculation (`SumOfMatrix`):**
   - Initialize sum variable to 0
   - Use nested loops to traverse entire matrix
   - Outer loop: iterate through rows (i = 0 to Rows-1)
   - Inner loop: iterate through columns (j = 0 to Cols-1)
   - Add each element `Matrix[i][j]` to the running sum
   - Return the total sum

3. **Display:** Print the original matrix and its calculated sum

### Step-by-step Process:
```
Matrix: 
 05  03  08 
 02  07  01 
 09  04  06 

Sum calculation:
5 + 3 + 8 + 2 + 7 + 1 + 9 + 4 + 6 = 45
```

### Mathematical Formula:
For an m×n matrix A, the sum is:
```
Sum = Σ(i=0 to m-1) Σ(j=0 to n-1) A[i][j]
```

### Use Cases:
- Statistical analysis (finding average by dividing sum by total elements)
- Matrix comparison based on total values
- Data validation and integrity checks

### Time Complexity: O(m×n) where m = rows, n = columns
### Space Complexity: O(1) - only storing the sum variable
