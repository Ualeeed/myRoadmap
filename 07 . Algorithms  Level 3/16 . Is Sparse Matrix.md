#include <iostream>
#include <string>
#include <iomanip>

using namespace std;


void PrintMatrix(int arr[3][3], short Rows, short Cols)
{
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			//printf(" %0*d ", 2, arr[i][j]);
			cout << setw(3) << arr[i][j] << " ";
		}
		cout << "\n";
	}
}


short CountNumberInMatrix(int Matrix1[3][3], int Number, short
	Rows, short Cols)
{
	short NumberCount = 0;
	for (short i = 0; i < Rows; i++)
	{
		for (short j = 0; j < Cols; j++)
		{
			if (Matrix1[i][j] == Number)
			{
				NumberCount++;
			};
		}
	}
	return NumberCount;
}


bool IsSparseMatrix(int Matrix1[3][3], short Rows, short Cols)
{
	short MatrixSize = Rows * Cols;

	return (CountNumberInMatrix(Matrix1, 0, 3, 3) >= ceil((float)
		MatrixSize / 2));
}


int main()
{

	int Matrix1[3][3] = { {0,0,12},{9,9,1},{0,0,9} }; 

	cout << "\nMatrix1:\n";

	PrintMatrix(Matrix1, 3, 3);

	/* sparse matrix is a matrix where zeros are more than 50 % of the numbers in matrix
	like ex : int Matrix1[3][3] = { {0,0,1},{0,0,1},{0,0,9} }; */

	if (IsSparseMatrix(Matrix1, 3, 3))
		cout << "\nYes: It is Sparse\n";
	else
		cout << "\nNo: It's NOT Sparse\n";

}

## Step-by-Step Algorithm Solution

### Algorithm: Check if Matrix is Sparse Matrix

**Problem:** Determine if a given matrix is sparse (contains more than 50% zeros).

**Steps:**
1. **Start** the algorithm
2. **Input:** Accept matrix dimensions (rows and columns) and matrix elements
3. **Calculate** total matrix size: `matrixSize = rows × columns`
4. **Count** the number of zeros in the matrix:
   - **Initialize** zeroCount to 0
   - **For each row i** from 0 to rows-1:
     - **For each column j** from 0 to columns-1:
       - **If** matrix[i][j] equals 0:
         - **Increment** zeroCount by 1
5. **Calculate** threshold: `threshold = matrixSize ÷ 2` (rounded up)
6. **Compare** zeroCount with threshold:
   - **If** zeroCount ≥ threshold:
     - **Return** true (matrix is sparse)
   - **Else**:
     - **Return** false (matrix is not sparse)
7. **Output** the result
8. **End** the algorithm

**Time Complexity:** O(n×m) where n is rows and m is columns
**Space Complexity:** O(1) - only using a few variables

**Key Points:**
- A sparse matrix has more than 50% of its elements as zeros
- Uses ceiling function to handle odd-sized matrices properly
- For a 3×3 matrix (9 elements), needs at least 5 zeros to be sparse
- Sparse matrices are common in scientific computing and are stored efficiently using special data structures
- The algorithm reuses the CountNumberInMatrix function for modularity

**Example:**
- Matrix: [[0,0,12], [9,9,1], [0,0,9]]
- Total elements: 9
- Zero count: 4
- Threshold: ceil(9/2) = 5
- Result: Not sparse (4 < 5)
