# Requirement : 

- Write a program that calculate loan Amount and Monthly payment 
# Solution :  


- my method : 

```cpp
#include <iostream>
#include <string>
using namespace std;
 
struct ReadfromUser
{
	int LoanAmount, MonthlyInstallment;
};
 
		
int ReadLoan()
{
	ReadfromUser User;

	cout << "Please enter LoanAmount ?" << endl;
	cin >> User.LoanAmount;

	return User.LoanAmount;
}

int ReadMonthly()
{
	ReadfromUser User;

	cout << "Please enter Monthly installment ?" << endl;
	cin >> User.MonthlyInstallment;

	return User.MonthlyInstallment;
}

int CalculateMonth()
{
	int TotalMonths = ReadLoan() / ReadMonthly();

	return TotalMonths;
}


int main()
{

	cout << CalculateMonth() << " Months" << endl;
 

	return 0;
}
```
