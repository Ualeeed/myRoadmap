# Requirement : 

- Write a program that calculate factorial of N! 
# Solution :  

- my method : 

```c++
#include <iostream>
using namespace std;

int ReadNumber()
{
	int N;

	cout << "Enter your Number : " << endl;
	cin >> N;
	 
	while(N <= 0)
	{
		cout << "Wrong Number, Please enter a positive number." << endl;
		cin >> N;
	}

	return N;
 }

void CalculateFactorial(int N)
{
	int counter = N ;
	float Factorial = 1;

	while (counter != 0)
	{
		Factorial *= counter;

		counter--;
	}

	cout << "Factorial of " << N << " = " << Factorial << endl;
}


int main()
{
	CalculateFactorial(ReadNumber());

	return 0;

}
```
