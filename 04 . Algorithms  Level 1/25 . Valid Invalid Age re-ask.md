# Requirement : 

- Write a program that check if the number is valid or not, if the number is Invalid `re_ask` the user to enter another number
# Solution :  

```cpp
#include <iostream>
#include <string>
using namespace std;

int ReadAge()
{
	int Age;

	cout << "Enter your Age please : " << endl;
	cin >> Age;

	return Age;
}

bool ValidateNumberInRange(int Age, int From, int To)
{
	return (Age >= From && Age <= To);
}

int ReadUntilAgeBetween(int From, int To)
{
	int Age = 0;
	 
	do
	{
		Age = ReadAge();

	} while (!ValidateNumberInRange(Age,From,To));

	return Age;
 
}

void PrintAge(int Age)
{
	cout << "Your Age is : " << Age << endl;
}


int main()
{

	 
	PrintAge(ReadUntilAgeBetween(18, 40));


	return 0;
}
```
