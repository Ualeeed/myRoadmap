# Requirement : 

- Write a program that calculate number of weeks based on number of hours
# Solution :   


my method : 

```cpp
#include <iostream>
#include <string>
using namespace std;

float NumberOfHours(string Message)
{
	float Hours = 0;

	cout << Message << endl;
	cin >> Hours;

	return Hours;
}

float CalculateDays(float Hours)
{
	float Days = Hours / 24;
	return Days;
}

float CalculateWeeks(float Hours)
{
	float Weeks = CalculateDays(Hours) / 7;
	return Weeks;
}
		
int main()
{
 
	float Hours = NumberOfHours("Please enter number of hours : ");

	cout << "Days = " << CalculateDays(Hours) << endl;
	cout << "**************************************** \n";
	cout << "Weeks = " << CalculateWeeks(Hours) << endl;


	return 0;
}
```


Abu-Hadhoud's method : 

```cpp
#include <iostream>
#include <string>
using namespace std;

float ReadPositiveNumber(string Message)
{
	float Number = 0;

	cout << Message << endl;
	cin >> Number;

	return Number;
}

float HoursToDays(float NumberOfHours)
{
	return NumberOfHours / 24;
}

float HoursToWeeks(float NumberOfHours)
{
	return NumberOfHours / 24 / 7;

}
float DaysToWeeks(float NumberOfDays)
{
	return  NumberOfDays  / 7;
}


		
int main()
{
	float NumberOfHours = ReadPositiveNumber("Please enter number of hours :");
	float NumberOfDays = HoursToDays(NumberOfHours);
	float NumberOfWeeks = DaysToWeeks(NumberOfDays);

	cout << endl;

	cout << "Total Hours = " << NumberOfHours << endl;
	cout << "Total Days = " << NumberOfDays << endl;
	cout << "Total Weeks = " << NumberOfWeeks << endl;
  


	return 0;
}
```
