# Requirement : 

- Write a program that print the day based on his number 
# Solution :  


my method : 

```cpp
#include <iostream>
#include <string>
using namespace std;

enum enWeekDays { Mon = 1, Tue = 2, Wed = 3, Thr = 4, Fri = 5, Sat = 6, Sun = 7 };

int ReadAndCheckDay(string Message)
{
	int Day;

	do
	{
		cout << Message << endl;
		cin >> Day;

	} while (!(Day <= 7 && Day >= 1));

	return Day;
 }


void CheckDay(int Day)
{
	

	if (Day == enWeekDays::Mon)
	
		cout << "\nMonday" << endl;
	
	else if (Day == enWeekDays::Tue)
	
		cout <<  "\nTuesday" << endl;
	
	else if (Day == enWeekDays::Wed)
	
		cout << "\nWednesday" << endl;
	
	else if (Day == enWeekDays::Thr)
	
		cout << "\nThursday" << endl;
	
	else if (Day == enWeekDays::Fri)
	
		cout << "\nFriday" << endl;
	
	else if (Day == enWeekDays::Sat)
	
		cout << "\nSaturday" << endl;
	
	else if (Day == enWeekDays::Sun)
	
		cout << "\nSunday" << endl;
	
}


		
int main()
{
	int Day = ReadAndCheckDay("Please enter the number of the Day 1 to 7 ?");
	 
	CheckDay(Day);

	return 0;
}
```

```cpp
#include <iostream>
#include <string>
using namespace std;

enum enDayOfWeek { Mon = 1, Tue = 2, Wed = 3, Thu = 4, Fri = 5, Sat = 6, Sun = 7 };

int ReadNumberInRange(string Message, int From,int To)
{
	int Number;

	do
	{
		cout << Message << endl;
		cin >> Number;

	} while (Number < From || Number > To);

	return Number;
 }


enDayOfWeek ReadDayOfWeek()
{
	
	return (enDayOfWeek)ReadNumberInRange("Please enter day number between 1 and 7 :", 1 , 7);
	 
}

string GetDayOfWeek(enDayOfWeek Day)
{
	switch (Day)
	{
	case enDayOfWeek::Mon:
		return "it's Monday";
		break;
	case enDayOfWeek::Tue:
		return "it's Tuesday";
		break;
	case enDayOfWeek::Wed:
		return "it's Wednesday";
		break;
	case enDayOfWeek::Thu:
		return "it's Thursday";
		break;
	case enDayOfWeek::Fri:
		return "it's Friday";

	case enDayOfWeek::Sat:
		return "it's Saturday";
		break;

		return "it's Sunday";

}
```
