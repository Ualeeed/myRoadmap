# Requirement : 

- Write a program that calculate task duration in seconds
# Solution :  

- my method : 
```cpp
#include <iostream>
#include <string>
using namespace std;

float ReadPositiveNumber(string Message)
{
	float Number;

	cout << Message << endl;
	cin >> Number;

	return Number;
 }

float DaysToSeconds(float NumberOfDays)
{
	return  (float)NumberOfDays * 24 * 60 * 60;
}

float HoursToSeconds(float NumberOfHours)
{
	return (float)NumberOfHours * 60 * 60;
}

float MinuteToSeconds(float NumberOfMinutes)
{
	return (float)NumberOfMinutes * 60;
}


		
int main()
{
	 
	float NumberOfDays = ReadPositiveNumber("Please enter number of days : ");
	float NumberOfHours = ReadPositiveNumber("Please enter number of Hours : ");
	float NumberOfMinutes = ReadPositiveNumber("Please enter number of Minutes : ");
	float NumberOfSeconds = ReadPositiveNumber("Please enter number of seconds : ");

	cout << endl;

	cout << "********************************************************************** \n";

	cout << "TotalSeconds = " << DaysToSeconds(NumberOfDays) + HoursToSeconds(NumberOfHours) + MinuteToSeconds(NumberOfMinutes) + NumberOfSeconds;

	return 0;
}
```

```cpp
#include <iostream>
#include <string>
using namespace std;

 struct strTaskDuration
 {
	 int NumberOfDays, NumberOfHours, NumberOfMinutes, NumberOfSeconds;
 };

 int ReadPositiveNumber(string Message)
 {
	 int Number = 0;
	 do
	 {
		 cout << Message << endl;
		 cin >> Number;

	 } while (Number <= 0);

	 return Number;
}

 strTaskDuration ReadTaskDuration()
 {
	 strTaskDuration TaskDuration;

	 TaskDuration.NumberOfDays = ReadPositiveNumber("Please enter number of days :");
	 TaskDuration.NumberOfHours = ReadPositiveNumber("Please enter number of Hours :");
	 TaskDuration.NumberOfMinutes = ReadPositiveNumber("Please enter number of Minutes :");
	 TaskDuration.NumberOfSeconds = ReadPositiveNumber("Please enter number of Seconds :");

	 return TaskDuration;
 }

 int TaskDurationInSeconds(strTaskDuration TaskDuration)
 {
	 int TotalSeconds;

	 return  TotalSeconds = (TaskDuration.NumberOfDays * 24 * 60 * 60) + (TaskDuration.NumberOfHours * 60 * 60) + (TaskDuration.NumberOfMinutes * 60) + TaskDuration.NumberOfSeconds;
 }
		
int main()
{
	cout << "\n Task Duration In Seconds :" << TaskDurationInSeconds(ReadTaskDuration());
	 
	return 0;
}
```
