# Requirement : 

- Write a program that calculate average and check based on average if the student PASS or FAIL
# Solution : 

```cpp
#include <iostream>
#include <string>
using namespace std;

enum enPassFail { Pass = 1, Fail = 2 };

void ReadNumber(int& Mark1, int& Mark2, int& Mark3)
{


	cout << "Enter Mark 1 : \n";
	cin >> Mark1;

	cout << "Enter Mark 2 : \n";
	cin >> Mark2;

	cout << "Enter Mark 3 : \n";
	cin >> Mark3;


}

int SumOf3Numbers(int Mark1, int Mark2, int Mark3)
{

	return Mark1 + Mark2 + Mark3;

}

float CalculateAverage(int Mark1, int Mark2, int Mark3)
{
	return (float)SumOf3Numbers(Mark1, Mark2, Mark3) / 3;
}

enPassFail TestPassFail(int Average)
{
	if (Average >= 50)
	{
		return enPassFail::Pass;
	}
	else
	{
		return enPassFail::Fail;
	}
}

void PrintSum(int Average)
{
	cout << "\n Your Average = " << Average << endl;
	
	if (TestPassFail(Average) == enPassFail::Pass)
	{
		cout << "\n You Passed " << endl;
	}
	else
	{
		cout << "\n You Failed " << endl;
	}
}


int main()
{
	int Mark1, Mark2, Mark3;

	ReadNumber(Mark1, Mark2, Mark3);
	PrintSum(CalculateAverage(Mark1, Mark2, Mark3));


	return 0;
}
```
