#include <iostream>
#include <string>
using namespace std;

 struct ReadFromUser
 {
     int Number1, Number2;
     char Optype;
 };

 ReadFromUser ReadUser()
 {
     ReadFromUser OperationAndNumber;

     cout << "enter N1 : \n";
     cin >> OperationAndNumber.Number1;
     cout << "enter N2 : \n";
     cin >> OperationAndNumber.Number2;
     cout << "enter Operation type (+ . - . * . / ) : \n";
     cin >> OperationAndNumber.Optype;

     return OperationAndNumber;
 }

 int CalculateNumbersByOpType(ReadFromUser OperationAndNumber)
 {
      

     if (OperationAndNumber.Optype == '+')

         return OperationAndNumber.Number1 + OperationAndNumber.Number2;

     else if (OperationAndNumber.Optype == '-')

         return OperationAndNumber.Number1 - OperationAndNumber.Number2;

     else if (OperationAndNumber.Optype == '/')

         return OperationAndNumber.Number1 / OperationAndNumber.Number2;
     else

         return OperationAndNumber.Number1 * OperationAndNumber.Number2;
      
     
 }
 

int main()
{
    cout << "Result = " << CalculateNumbersByOpType(ReadUser());

    return 0;
}
