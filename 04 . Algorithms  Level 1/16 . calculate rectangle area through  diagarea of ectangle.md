# Requirement : 

- Write a program that calculate rectangle area through diagonal and side area of rectangle.
# Solution :   



```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadNumbers(float& A, float& D)
{
	cout << "Please enter rectangle side A : " << endl;
	cin >> A;

    cout << "Please enter rectangle diagonal D : " << endl;
	cin >> D;
 }
 
float RectangleAreaBySideAndDiagonal(float A, float D)
{
	float Area;
	return  Area = A * sqrt(pow(D, 2) - pow(A, 2));

}
 
void PrintResult(float Area)
{
	cout << "\n Rectangle Area by side and diagonal = " << Area << endl;
}
 
int main()
{
	float A, D;

	ReadNumbers(A, D);
	PrintResult(RectangleAreaBySideAndDiagonal(A, D));


	return 0;
}
```
