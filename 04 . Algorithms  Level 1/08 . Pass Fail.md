# Requirement : 

- Write a program, to take the mark from the user and check based on the mark if it's PASS or FAIL 
# Solution : 

```cpp
#include <iostream>
#include <string>
using namespace std;

enum enPassFail { Pass =0, Fail=2 };

int ReadMark()
{
	int Mark;

	cout << "Enter your Mark " << endl;
	cin >> Mark;

	return Mark;
}

enPassFail CheckMark(int Mark)
{

	if (Mark >= 50)
	{
		return enPassFail::Pass;
	}
	else
	{
		return enPassFail::Fail;
	}

}

void PrintResult(int Mark)
{
	
	if (CheckMark(Mark) == enPassFail::Pass)
	{
		cout << " \n PASS" << endl;
	}
	else
	{
		cout << " \n FAIL" << endl;
	}
}

int main()
{
	PrintResult(ReadMark());

	return 0;
}
```
