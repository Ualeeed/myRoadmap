# Requirement : 

- Write a program that calculate sum even numbers from 1 to N 
# Solution :  

```cpp
#include <iostream>
using namespace std;

enum enOddEven {Odd=1, Even=2};

int ReadNumber()
{
	int N;
	cout << "Please Enter N : " << endl;
	cin >> N;

	return N;
}

enOddEven CheckOddEven(int Number)
{

	if (Number % 2 == 0)
	
		return enOddEven::Even;
	
	else
	
		return enOddEven::Odd;
	
}

void SumOddNumbersFrom1toN_UsingWhile(int N)
{
	int counter = 0;
	int Sum = 0;

	cout << "Using While Loop" << endl;

	while (counter < N )
	{
		counter++;

		if (CheckOddEven(counter) == enOddEven::Even)
		{
			Sum += counter;
		}
		
	}

	cout << Sum << endl;
}

void SumOddNumbersFrom1toN_UsingDoWhile(int N)
{
	int counter = 0;
	int Sum = 0;

	cout << "Using Do..While Loop" << endl;

	do
	{
		counter++;
		
		if (CheckOddEven(counter) == enOddEven::Even)
		{
			Sum += counter;

		}
		

	} while (counter < N);

	cout << Sum << endl;

}

void SumOddNumbersFrom1toN_UsingFor(int N)
{
	int Sum = 0;

	cout << "Using For Loop" << endl;


	for (int counter = 1; counter <= N; counter++)
	{
		if (CheckOddEven(counter) == enOddEven::Even)
		{
			Sum += counter;
		}

	}

	cout << Sum << endl;
}

int main()
{
	int N = ReadNumber();

	 
	SumOddNumbersFrom1toN_UsingWhile(N);

	SumOddNumbersFrom1toN_UsingDoWhile(N);

	SumOddNumbersFrom1toN_UsingFor(N);

	return 0;


}
```
