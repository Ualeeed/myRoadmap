# Requirement : 

- Write a program that calculate rectangle are 
# Solution : 

```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadNumber(float& A, float& B)
{
	cout << "Please enter rectangle with : " << endl;
	cin >> A;

	cout << "Please enter rectangle Length : " << endl;
	cin >> B;
}
 
float CalculateArea(float A, float B)
{
	return A * B;
}

void PrintCalculateArea(float Area)
{
	cout << "\n Rectangle Area = " << Area << endl;
}
 
int main()
{
	float A, B;

	ReadNumber(A, B);
	PrintCalculateArea(CalculateArea(A, B));

	return 0;
}
```
