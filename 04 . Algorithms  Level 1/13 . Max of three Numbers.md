# Requirement : 

- Write a program that check the max of the three numbers
# Solution : 

```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadNumbers(int& A, int& B, int &C)
{
	cout << "Enter Number 1 :" << endl;
	cin >> A;

	cout << "Enter Number 2 :" << endl;
	cin >> B;

	cout << "Enter Number 3 :" << endl;
	cin >> C;
}

int MaxOf3Numbers(int A, int B, int C)
{
	if (A > B && A > C)
	{
		return A;
	}
	else if (B > A && B > C)
	{
		return B;
	}
	else
	{
		return C;
	}
}

void PrintMaxNumber(int Max)
{
	cout << "\n Max Number is : " << Max << endl;
}

int main()
{
	int A, B, C;

	ReadNumbers(A, B, C);
	PrintMaxNumber(MaxOf3Numbers(A, B, C));

	return 0;
}
```
