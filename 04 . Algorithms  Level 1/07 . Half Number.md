# Requirement : 

- Write a program to read a number from the user then print the half of its 
# Solution : 

```cpp
#include <iostream>
#include <string>
using namespace std;

int ReadNumber()
{
	int Number;

	cout << "Enter a number : \n ";
	cin >> Number;

	return Number;
}

float CalculateHalfOfNumber(int Number)
{

	return (float)Number / 2;

}

void PrintResult(int Number)
{
	string Result = "Half of " + to_string(Number) + " is " + to_string(CalculateHalfOfNumber(Number));

	cout << Result << endl;

}



int main()
{
	PrintResult(ReadNumber());
 

	return 0;
}
```
