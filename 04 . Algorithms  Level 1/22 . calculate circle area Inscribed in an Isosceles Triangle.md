# Requirement : 

- Write a program that calculate circle area inscribed in an isosceles triangle 
# Solution :  

```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadFromUser(float& a, float& b)
{

	cout << "Please enter a  : " << endl;
	cin >> a;

	cout << "Please enter b  : " << endl;
	cin >> b;

}
 
float CircleAreaInscribedInAnIsoscelesTriangle(float a, float b)
{
	
	const float PI = 3.141592653589783238;

	float Area = (PI * pow(b, 2) / 4) * (2 * a - b) / (2 * a + b);

	return Area;

}
 
void PrintResult(float Area)
{
	
	cout << "\n Circle Area Inscribed In An Isosceles Triangle  = " << Area << endl;
}
 
int main()
{
	float a, b;
	ReadFromUser(a, b);

	PrintResult(CircleAreaInscribedInAnIsoscelesTriangle(a, b));
	return 0;
}
```
