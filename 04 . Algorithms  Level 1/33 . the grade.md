# Requirement : 

- Write a program that calculate Grade
# Solution :  

- my method :

```cpp
#include <iostream>
#include <string>
using namespace std;

 
int ReadGrade()
{
	int Grade;

	cout << "Please enter your Grade : " << endl;
	cin >> Grade;

	return Grade;
}

char CheckGrade(int Grade)
{
	

	if (Grade >= 90)
	{
		return 'A';
	}
	else if (Grade >= 80)
	{
		return 'B';
	}
	else if (Grade >= 70)
	{
		return 'C';
	}
	else if (Grade >= 60)
	{
		return 'D';
	}
	else if (Grade >= 50)
	{
		return 'E';
	}
	else 
	{
		return 'F';
	}


 }

void PrintResult(int Grade)
{
	cout << "Your Mark is : " << CheckGrade(Grade) << endl;
}

int main()
{
	
 
	PrintResult(ReadGrade());
	
 

	return 0;

}
```

- Abu-Hadhoud's method : 

```cpp
#include <iostream>
#include <string>
using namespace std;

 
int ReadNumberInRange(int From, int To)
{
	int Grade;

	do
	{
		cout << "Please enter your grade between 0 and 100 : " << endl;
		cin >> Grade;

	} while (Grade < From || Grade > To);

	return Grade;
}

char GetGradeLetter(int Grade)
{
	if (Grade >= 90)
		return 'A';
	else if (Grade >= 80)
		return 'B';
	else if (Grade >= 70)
		return 'C';
	else if (Grade >= 60)
		return 'D';
	else if (Grade >= 50)
		return 'E';
	else 
		return 'F';

 }

int main()
{
	
 
	cout << "Result = " << GetGradeLetter(ReadNumberInRange(0, 100)) << endl;
	


	return 0;

}
```
