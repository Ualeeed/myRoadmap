# Requirement : 

- Write a program that calculate circle area
# Solution :   

```cpp
#include <iostream>
#include <string>
using namespace std;

float ReadRadious()
{
	float r;

	cout << "Please enter radious r : " << endl;

	cin >> r;

	return r;
 }
 
float CircleArea(float r)
{
	
	const float PI = 3.141592653589783238;

	float Area = pow(r, 2) * PI;

	return Area;

}
 
void PrintResult(float Area)
{
	
	cout << "\n  Circle Area = " << Area << endl;
}
 
int main()
{
	PrintResult(CircleArea(ReadRadious()));


	return 0;
}
```
