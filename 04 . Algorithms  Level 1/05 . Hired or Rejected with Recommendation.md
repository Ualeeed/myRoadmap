# Requirement : 

- Write a program check info but this time check if the employee has a recommendation : 
# Solution : 


```cpp
#include <iostream>
#include <string>

using namespace std;

struct stInfo
{
	int Age;
	bool HasDriverLicense;
	bool HasRecommendation;
 };

stInfo ReadInfo()
{

	stInfo Info;

	cout << "Enter Your age : \n";
	cin >> Info.Age;

	cout << "De you have a driver license ? \n";
	cout << " (0) : No \n";
	cout << " (1) : Yes \n";

	cin >> Info.HasDriverLicense;

	cout << "De you have any Recommendation ? \n";
	cout << "(0) : No \n";
	cout << "(1) : Yes \n";

	cin >> Info.HasRecommendation;

	return Info;
}

bool IsAccepted(stInfo Info)
{
	  if (Info.HasRecommendation)
	{
		  return true;
	} 
	else
	{
		  return (Info.Age > 21 && Info.HasDriverLicense);
	}
}

void PrintResult(stInfo Info)
{
	if (IsAccepted(Info))
	{
		cout << "YOU ARE HIRED " << endl;
	} 
	else
	{
		cout << "YOU ARE REJECTED " << endl;
	}
}

  

int main()
{
 
	PrintResult(ReadInfo());


}
```
