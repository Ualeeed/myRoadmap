# Requirement : 

- Write a program that print all letters from A to Z 
# Solution :   
  
```cpp
#include <iostream>
#include <string>

using namespace std;


void PrintLetterAtoZ()
{
	for (int i = 65; i <= 90; i++)
	{
		cout << char(i) << endl;
	}
 }
		
int main()
{

	PrintLetterAtoZ();
 

	return 0;
}
```
