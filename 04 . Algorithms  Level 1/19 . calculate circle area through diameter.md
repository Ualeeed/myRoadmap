# Requirement : 

- Write a program that calculate circle area through diameter
# Solution :

```cpp
#include <iostream>
#include <string>
using namespace std;

float ReadDiameter()
{
	float D;

	cout << "Please enter diameter  : " << endl;

	cin >> D;

	return D;
 }
 
float CircleAreaByDiameter(float D)
{
	
	const float PI = 3.141592653589783238;

	float Area = (PI * pow(D, 2)) / 4;

	return Area;

}
 
void PrintResult(float Area)
{
	
	cout << "\n  Circle Area through diameter = " << Area << endl;
}
 
int main()
{
	
	PrintResult(CircleAreaByDiameter(ReadDiameter()));

	return 0;
}
```
