# Requirement : 

- Write a program that calculate sum of three numbers
# Solution : 

```cpp

#include <iostream>
#include <string>
using namespace std;

void ReadNumber(int &Num1, int &Num2, int &Num3)
{
	

	cout << "Enter Number 1 : \n";
	cin >> Num1;

	cout << "Enter Number 2 : \n";
	cin >> Num2;

	cout << "Enter Number 3 : \n";
	cin >> Num3;

	
}

int SumOf3Numbers(int Num1, int Num2, int Num3)
{

	return Num1 + Num2 + Num3;

}

void PrintSum(int Total)
{
	cout << Total << endl;
}
int main()
{
	int Num1, Num2, Num3;

	ReadNumber(Num1, Num2, Num3);
	PrintSum(SumOf3Numbers(Num1, Num2, Num3));
	 

	return 0;
}

```
