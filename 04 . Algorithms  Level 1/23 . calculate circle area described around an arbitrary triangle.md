# Requirement : 

- Write a program that calculate circle area described around an arbitrary triangle 
# Solution :  

```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadFromUser(float& a, float& b, float& c)
{

	cout << "Please enter Triangle side a  : " << endl;
	cin >> a;

	cout << "Please enter Triangle side b  : " << endl;
	cin >> b;

	cout << "Please enter Triangle side c  : " << endl;
	cin >> c;
}
 
float CircleAreaDescribedAroundAnArbitraryTringle(float a, float b, float c)
{
	
	const float PI = 3.141592653589783238;
	float p = (a + b + c) / 2;

	float T = (a * b * c) / (4 * sqrt(p * (p - a) * (p - b) * (p - c)));
	
	float Area = PI * pow(T, 2);

	return Area;

}
 
void PrintResult(float Area)
{
	
	cout << "\n Circle Area   = " << Area << endl;
}
 
int main()
{
	float a, b, c;
	ReadFromUser(a, b, c);

	PrintResult(CircleAreaDescribedAroundAnArbitraryTringle(a, b, c));

	return 0;
}
```
