# Requirement : 

- Write a program that print month based on his number 
# Solution :  

```cpp
#include <iostream>
#include <string>
using namespace std;

enum enMonthOfYear { Jan = 1, Feb = 2, Mar = 3, Apr = 4, May = 5, June = 6, July = 7, Aug = 8, Sep = 9, Oct = 10, Nov = 11, Dec = 12 };

int ReadNumberInRange(string Message, int From,int To)
{
	int Number;

	do
	{
		cout << Message << endl;
		cin >> Number;

	} while (Number < From || Number > To);

	return Number;
 }


enMonthOfYear ReadMonthOfYear()
{
	
	return (enMonthOfYear)ReadNumberInRange("Please enter month number between 1 and 12 :", 1 , 12);
	 
}

string GetMonthOfYear(enMonthOfYear Month)
{
	if (Month == enMonthOfYear::Jan)
	{
		return "it's  January ";
	}
	else if (Month == enMonthOfYear::Feb)
	{
		return "it's February";
	}
	else if (Month == enMonthOfYear::Mar)
	{
		return "it's March";
	}
	else if (Month == enMonthOfYear::Apr)
	{
		return "it's April";
	}
	else if (Month == enMonthOfYear::May)
	{
		return "it's May";
	}
	else if (Month == enMonthOfYear::June)
	{
		return "it's June";
	}
	else if (Month == enMonthOfYear::July)
	{
		return "it's July";
	}
	else if (Month == enMonthOfYear::Aug)
	{
		return "it's August";
	}
	else if (Month == enMonthOfYear::Sep)
	{
		return "it's September";
	}
	else if (Month == enMonthOfYear::Oct)
	{
		return "it's October";
	}
	else if (Month == enMonthOfYear::Nov)
	{
		return "it's November";
	}
	else if (Month == enMonthOfYear::Dec)
	{
		return "it's December";
	}
	else
	{
		return "Wrong Month";
	}
}


		
int main()
{

	cout << GetMonthOfYear(ReadMonthOfYear()) << endl;

	return 0;
}
```
