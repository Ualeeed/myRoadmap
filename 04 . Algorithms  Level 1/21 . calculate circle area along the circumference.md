# Requirement : 

- Write a program that calculate circle area along the circumference 
# Solution :  

```cpp
#include <iostream>
#include <string>
using namespace std;

float ReadDiameter()
{
	float L;

	cout << "Please enter squire side L  : " << endl;

	cin >> L;

	return L;
 }
 
float CircleAreaAlongTheCircumference(float L)
{
	
	const float PI = 3.141592653589783238;

	float Area = pow(L, 2) / (4 * PI);

	return Area;

}
 
void PrintResult(float Area)
{
	
	cout << "\n  Circle Area along the circumference = " << Area << endl;
}
 
int main()
{
	
	PrintResult(CircleAreaAlongTheCircumference(ReadDiameter()));

	return 0;
}
```
