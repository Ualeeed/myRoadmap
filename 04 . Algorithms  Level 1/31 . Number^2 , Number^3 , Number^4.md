# Requirement : 

- Write a program that calculate Number^2 , Number^3 and Number^4 
# Solution :  

- my method : 

```cpp
#include <iostream>
#include <cmath>
using namespace std;

int ReadNumber()
{
	int Number;

	cout << "Please enter a Number : " << endl;
	cin >> Number;

	return Number;
	
}

int CalculatePow2(int Number)
{
	return pow(Number, 2);
}

int CalculatePow3(int Number)
{
	return pow(Number, 3);
}

int CalculatePow4(int Number)
{
	return pow(Number, 4);
}

void PrintCalculatedPow(int Number)
{

	cout << Number << "^2 = " << CalculatePow2(Number) << endl;

	cout << Number << "^3 = " << CalculatePow3(Number) << endl;

	cout << Number << "^4 = " << CalculatePow4(Number) << endl;

}

int main()
{
	
	PrintCalculatedPow(ReadNumber());

	return 0;

}
```

#include <iostream>
#include <cmath>
using namespace std;

int ReadNumber()
{
	int Number;

	cout << "Please enter a Number : " << endl;
	cin >> Number;

	return Number;
	
}



void PrintPow_2_3_4(int Number)
{
	int a, b, c;

	a = Number * Number;
	b = Number * Number * Number;
	c = Number * Number * Number * Number;

	cout << a << " " <<  b << " " << c << endl;

}

int main()
{
	
	PrintPow_2_3_4(ReadNumber());

	return 0;

}
