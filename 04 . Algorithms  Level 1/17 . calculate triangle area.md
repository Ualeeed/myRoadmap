# Requirement : 

- Write a program that calculate triangle area
# Solution :   


```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadNumbers(float& A, float& H)
{
	cout << "Please enter Triangle base : " << endl;
	cin >> A;

    cout << "Please enter Triangle height : " << endl;
	cin >> H;
 }
 
float TriangleArea(float A, float H)
{
	float Area = (A / 2) * H;
	return  Area;

}
 
void PrintResult(float Area)
{
	cout << "\n Triangle Area = " << Area << endl;
}
 
int main()
{
	float A, H;

	ReadNumbers(A, H);
	PrintResult(TriangleArea(A, H));


	return 0;
}
```
