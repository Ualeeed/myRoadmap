# Requirement : 

- Write a program that calculate circle area inscribed in a square
# Solution :   

```cpp
#include <iostream>
#include <string>
using namespace std;

float ReadDiameter()
{
	float A;

	cout << "Please enter squire side A  : " << endl;

	cin >> A;

	return A;
 }
 
float CircleAreaInscribedInaSquare(float A)
{
	
	const float PI = 3.141592653589783238;

	float Area = (PI * pow(A, 2)) / 4;

	return Area;

}
 
void PrintResult(float Area)
{
	
	cout << "\n  Circle Area inscribed in a square = " << Area << endl;
}
 
int main()
{
	
	PrintResult(CircleAreaInscribedInaSquare(ReadDiameter()));

	return 0;
}
```
