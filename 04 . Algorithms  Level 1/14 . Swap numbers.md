# Requirement : 

- Write a program that swap numbers place
# Solution : 

```cpp
#include <iostream>
#include <string>
using namespace std;

void ReadNumber(int& A, int& B)
{
	cout << "Enter Number1 : " << endl;
	cin >> A;

	cout << "Enter Number2 : " << endl;
	cin >> B;
}

void Swap(int& A, int& B)
{
	int Temp;

	Temp = A;
	A = B;
	B = Temp;

}

void PrintNumbers(int A, int B)
{
	cout << "\n Number1 = " << A << endl;
	cout << " Number2 = " << B << endl;

	cout << '\n';
}
int main()
{
	int Num1, Num2;

	ReadNumber(Num1, Num2);
	PrintNumbers(Num1, Num2);
	Swap(Num1, Num2);
	PrintNumbers(Num1, Num2);
	

	return 0;
}
```
