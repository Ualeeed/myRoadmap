#### Polymorphism

- polymorphism means "many forms"
- polymorphism allows us to create consistent code.

##### Polymorphism can be done through : 

1. **Function Overloading : 

- Example of the function overloading 

```cpp
#include <iostream>
using namespace std;

int Sum(int a, int b)
{
 return a + b;
}

int Sum(int a, int b, int c)
{
 return a + b + c;
}

int Sum(int a, int b, int c, int d)
{
 return a + b + c + d;
}

int main()
{

     cout << Sum(10, 20) << endl;
     cout << Sum(10, 20, 30) << endl;
     cout << Sum(10, 20, 30, 40) << endl;

 return 0;
}
```


2. **Operator Overloading : 

- Example of the Operator overloading

```cpp
#include <iostream>
using namespace std;

int main()
{

// this is what we called operator overloading iin this case (+)
    int x = 10;
    int y = 20;

    cout << x + y << endl;

    string FirstName = "Walid";
    string LastName = "Ahlidou";

    cout << FirstName + " " + LastName << endl;

     return 0;
}

```



3. **Function Overriding : 

- Example of the function overloading 

```cpp
#include <iostream>
using namespace std;

class clsA
{

public:
    void print()
    {
        cout << "Hi, I'm Function A" << endl;
    }
};

class clsB : clsA
{

public:
    void print()
    {
        cout << "Hi, I'm Function B" << endl;
    }
};

class clsC : clsB
{

public:
    void print()
    {
        cout << "Hi, I'm Function C" << endl;
    }
};

int main()
{
    clsA F1;
    F1.print();

    clsB F2;
    F2.print();

    clsC F3;
    F3.print();

    return 0;
}

```



1. **Virtual Functions : 

- Example of the virtual function 

```cpp
#include <iostream>
using namespace std;

class clsA
{

public:
    virtual void print()
    {
        cout << "Hi, I'm Function A" << endl;
    }
};

class clsB : public clsA
{

public:
    void print()
    {
        cout << "Hi, I'm Function B" << endl;
    }
};

class clsC : public clsA
{

public:
    void print()
    {
        cout << "Hi, I'm Function C" << endl;
    }
};

int main()
{

    clsB F2;
    clsC F3;

    clsA *F1 = &F2;
    clsA *F4 = &F3;

    F1->print();
    F4->print();

    return 0;
}
```
